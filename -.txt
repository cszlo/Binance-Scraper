#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main()
{
	char input[20] = { '1','2','3','U','C','F','1','2','3','4',' ','1','2','U','C','F','1','2','3','4' };
	char champs[38] = {
		'U', 'n', 'd', 'e', 'f', 'e', 'a', 't', 'e', 'd', ' ',
		'2', '0', '1', '8', ' ',
		'N', 'a', 't', 'i', 'o', 'n', 'a', 'l', ' ',
		'C', 'h', 'a', 'm', 'p', 'i', 'o', 'n', 's', ' ',
		'U', 'C', 'F'
	};

	int inputsize = sizeof(input) / sizeof(input[0]);
	int *indicies = malloc(sizeof(int) * inputsize + 1);

	int k;
	int i = 0;
	int count = 0;
	int j = 0;
	for (i = 0; i < inputsize; i++) {
		indicies[i] = 0;
	}

	for (i = 0; i < inputsize - 2; i++) {
		if (input[i] == 'U') {
			if (input[i + 1] == 'C') {
				if (input[i + 2] == 'F') {
					indicies[i] = i;
					count++;
				}
			}
		}
	}

	int newMSGsize = ((38 * count) + inputsize) - ((3 * count) + 1);
	char *newMsg = malloc(sizeof(char)* newMSGsize + 1);

	i = 0;
	while (i < inputsize && j < newMSGsize) {
		if (indicies[i] == 0) {
			newMsg[j] = input[i];
			i++;
			j++;
			newMsg[j + 1] = '\0';

		}
		else {
			for (k = 0; k < 38; k++, j++) {
				newMsg[j] = champs[k];
				newMsg[j + 1] = '\0';
			}
			i = indicies[i] + 3;
		}
	}
	printf("%s\n", newMsg);
	system("pause");
	return 0;
}
